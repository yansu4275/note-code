默认点击显示

				//基本使用, 默认当前年份加十年，不显示秒
				$("#input1").shijian();

直接在页面显示

				//直接在页面显示 
				$("#input2").shijian({
					showNowTime:false,
					Year:false,//是否显示年
					Month:false,//是否显示月
					Day:false,//是否显示日
					h:"00",
					m:"01",
					alwaysShow:true,
					timeElm:$(".time-box"),//sjObj表示当前对象  也可以使用 sjObj  用字符串 
					//timeElm:"sjObj.prev()",// 也可以使用 sjObj  用字符串  sjObj 表示当前input
				})

选择需要显示的

				//选择需要显示的 
				$("#input3").shijian({
					y:-10,//当前年份+10 
					Hour:false,//是否显示小时
				 	Minute:false,//是否显分钟
				})

自定义起始年份 

				//可直接在 input 的 value 填写默认展示的时间， 不过需要在指定范围内
				 $("#input4").shijian({
				 	startYear:1989,
				 	val:1989,
				 	endYear:2017,
				 	Hour:false,//是否显示小时
				 	Minute:false,//是否显分钟
				})
